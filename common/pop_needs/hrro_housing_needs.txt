# Housing Mod - Pop Needs Definition

# 租房需求 —— 使用单一商品，无替换
# 曲线类型: Temporary Need（中低财富出现，高财富消退）
popneed_rented_housing = {
    default = rented_housing
    entry = {
        goods = rented_housing
        weight = 1
        max_supply_share = 1.0
        min_supply_share = 0.0
    }
}

# 产权房需求 —— 使用单一商品，无替换
# 曲线类型: Exponential Need（高财富出现，指数增长）
popneed_owned_housing = {
    default = owned_housing
    entry = {
        goods = owned_housing
        weight = 1
        max_supply_share = 1.0
        min_supply_share = 0.0
    }
}
